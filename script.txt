function toggleWO(i) {
  const exp = document.getElementById('wo-exp-'+i);
  const item = exp.closest('.tli');
  const was = exp.classList.contains('show');
  document.querySelectorAll('.tli-expand.show').forEach(e=>{e.classList.remove('show');e.closest('.tli').classList.remove('open');});
  if(!was){exp.classList.add('show');item.classList.add('open');}
}

function setWO(btn,mode) {
  document.querySelectorAll('#page-watchorder .tog').forEach(b=>b.classList.remove('on'));
  btn.classList.add('on');
  woMode=mode; applyWO();
}
function setEra(el,era) {
  document.querySelectorAll('.epill').forEach(e=>e.classList.remove('on'));
  el.classList.add('on');
  woEra=era; applyWO();
}
function applyWO() {
  let items = [...MOVIES];
  if(woMode==='monsterverse') items=items.filter(m=>m.universe==='monsterverse');
  else if(woMode==='toho') items=items.filter(m=>m.universe==='toho');
  else if(woMode==='animation') items=items.filter(m=>m.universe==='animation');
  if(woEra!=='all') items=items.filter(m=>m.era===woEra);
  renderWO(items);
}

function filterTdex() {
  const q = document.getElementById('tdex-search').value.toLowerCase();
  const al = document.getElementById('tdex-align').value;
  const st = document.getElementById('tdex-status').value;
  const th = document.getElementById('tdex-threat').value;
  const un = document.getElementById('tdex-era').value;
  renderTdex(TITANS.filter(t=>{
    const mq=!q||[t.name,t.alias,t.origin,...t.powers,...(t.tags||[])].join(' ').toLowerCase().includes(q);
    const ma=al==='all'||t.align===al;
    const ms=st==='all'||t.status===st;
    const mt=th==='all'||t.threat===th;
    const mu=un==='all'||t.universe===un||t.universe==='both';
    return mq&&ma&&ms&&mt&&mu;
  }));
}

// RELATIONSHIP WEB (Canvas)
// ═══════════════════════════════════════════════════════════
let relNodes=[], relDragging=null, relMouse={x:0,y:0}, relAnimating=false;

function renderRelWeb() {
  const canvas = document.getElementById('rel-canvas');
  if(!canvas) return;
  const W=canvas.offsetWidth, H=canvas.offsetHeight;
  canvas.width=W; canvas.height=H;
  const ctx=canvas.getContext('2d');

  // Build unique nodes
  const names = [...new Set(RELATIONS.flatMap(r=>[r.from,r.to]))];
  const cx=W/2, cy=H/2, R=Math.min(W,H)*0.36;
  
  relNodes = names.map((n,i)=>{
    const angle=(i/names.length)*Math.PI*2 - Math.PI/2;
    const t = TITANS.find(tt=>tt.name===n);
    return {
      name:n, emoji:t?t.emoji:'❓',
      x:cx+Math.cos(angle)*R*(0.7+Math.random()*0.3),
      y:cy+Math.sin(angle)*R*(0.7+Math.random()*0.3),
      vx:0, vy:0, r:28
    };
  });

  // Physics simulation
  function simulate() {
    for(let i=0;i<relNodes.length;i++){
      for(let j=i+1;j<relNodes.length;j++){
        const dx=relNodes[j].x-relNodes[i].x, dy=relNodes[j].y-relNodes[i].y;
        const dist=Math.sqrt(dx*dx+dy*dy)||1;
        const force=8000/(dist*dist);
        relNodes[i].vx-=dx/dist*force*0.016;
        relNodes[i].vy-=dy/dist*force*0.016;
        relNodes[j].vx+=dx/dist*force*0.016;
        relNodes[j].vy+=dy/dist*force*0.016;
      }
    }
    RELATIONS.forEach(r=>{
      const a=relNodes.find(n=>n.name===r.from), b=relNodes.find(n=>n.name===r.to);
      if(!a||!b) return;
      const dx=b.x-a.x, dy=b.y-a.y;
      const dist=Math.sqrt(dx*dx+dy*dy)||1;
      const target=140;
      const f=(dist-target)*0.02;
      a.vx+=dx/dist*f; a.vy+=dy/dist*f;
      b.vx-=dx/dist*f; b.vy-=dy/dist*f;
    });
    relNodes.forEach(n=>{
      if(n===relDragging) return;
      n.x+=n.vx; n.y+=n.vy;
      n.vx*=0.85; n.vy*=0.85;
      n.x=Math.max(n.r,Math.min(W-n.r,n.x));
      n.y=Math.max(n.r,Math.min(H-n.r,n.y));
    });
  }

  const typeColors = {ally:'#00ff88',enemy:'#ff4d00',alpha:'#e8b400',neutral:'#00aaff'};

  function draw() {
    ctx.clearRect(0,0,W,H);
    // Edges
    RELATIONS.forEach(r=>{
      const a=relNodes.find(n=>n.name===r.from), b=relNodes.find(n=>n.name===r.to);
      if(!a||!b) return;
      ctx.beginPath();
      ctx.moveTo(a.x,a.y); ctx.lineTo(b.x,b.y);
      ctx.strokeStyle=typeColors[r.type]||'#555';
      ctx.lineWidth=r.type==='ally'?2:1.5;
      ctx.globalAlpha=0.45;
      ctx.stroke();
      ctx.globalAlpha=1;
    });
    // Nodes
    relNodes.forEach(n=>{
      ctx.beginPath();
      ctx.arc(n.x,n.y,n.r,0,Math.PI*2);
      ctx.fillStyle='#0a1520';
      ctx.fill();
      ctx.strokeStyle='#1e3a55';
      ctx.lineWidth=1.5;
      ctx.stroke();
      ctx.font=`${n.r*0.9}px serif`;
      ctx.textAlign='center';
      ctx.textBaseline='middle';
      ctx.fillText(n.emoji,n.x,n.y);
      ctx.font='500 10px Rajdhani,sans-serif';
      ctx.fillStyle='#c0d8e8';
      ctx.fillText(n.name,n.x,n.y+n.r+11);
    });
  }

  function loop() {
    if(!relAnimating) return;
    simulate(); draw();
    requestAnimationFrame(loop);
  }

  // Drag
  canvas.onmousedown=e=>{
    const rect=canvas.getBoundingClientRect();
    const mx=e.clientX-rect.left, my=e.clientY-rect.top;
    relDragging=relNodes.find(n=>Math.hypot(n.x-mx,n.y-my)<n.r+4)||null;
  };
  canvas.onmousemove=e=>{
    if(!relDragging) return;
    const rect=canvas.getBoundingClientRect();
    relDragging.x=e.clientX-rect.left;
    relDragging.y=e.clientY-rect.top;
    relDragging.vx=0; relDragging.vy=0;
  };
  canvas.onmouseup=()=>{relDragging=null;};

  relAnimating=true;
  loop();
}

  [1,2].forEach(s=>{

    slot.innerHTML=t?`<div class="sim-emoji-big">${t.emoji}</div><div class="sim-fname">${t.name}</div>

    `<div class="sim-placeholder">+ SELECT TITAN ${s}</div>`;
    slot.classList.toggle('selected',!!t);
  });
  document.getElementById('fight-btn').disabled=!(simFighter[1]&&simFighter[2]);
  document.getElementById('sim-result').classList.remove('show');
}

function runSim() {
  const a=simFighter[1], b=simFighter[2];
  if(!a||!b) return;
  // Scoring system
  const threatScore={low:1,medium:3,high:6,omega:10};
  const alignBonus={benevolent:1,neutral:0,alpha:1,hostile:-0.5,unknown:0};
  const statusPenalty={alive:0,dormant:-2,deceased:-10,unknown:-1};
  
  function score(t) {
    let s=threatScore[t.threat]||5;
    s+=t.powers.length*0.3;
    s+=alignBonus[t.align]||0;
    s+=statusPenalty[t.status]||0;
    s+=Math.random()*2; // variance
    return s;
  }
  
  const sa=score(a), sb=score(b);
  const winner=sa>=sb?a:b;
  const loser=sa>=sb?b:a;
  const margin=Math.abs(sa-sb);
  const close=margin<2;
  
  const analyses=[
    `${winner.name}'s ${winner.powers[0]} proves decisive, overpowering ${loser.name}'s defenses.`,
    `The battle rages across multiple terrains. ${winner.name} endures. ${loser.name} ultimately withdraws.`,
    `${winner.name}'s superior ${winner.threat === 'omega'?'omega-level energy output':'combat adaptability'} turns the tide.`,
    `${loser.name} staggers ${winner.name} early — but ${winner.name}'s regeneration and ${winner.powers[Math.floor(Math.random()*winner.powers.length)]} prove insurmountable.`,
  ];
  const analysis=analyses[Math.floor(Math.random()*analyses.length)];

  const statRows=[
    {label:"THREAT",a:a.threat.toUpperCase(),b:b.threat.toUpperCase(),aWin:threatScore[a.threat]>=threatScore[b.threat]},
    {label:"ABILITIES",a:a.powers.length,b:b.powers.length,aWin:a.powers.length>=b.powers.length},
    {label:"STATUS",a:a.status.toUpperCase(),b:b.status.toUpperCase(),aWin:statusPenalty[a.status]>=statusPenalty[b.status]},
  ];

  const res=document.getElementById('sim-result');
  res.innerHTML=`
    <div class="res-title">${close?'CLOSE BATTLE':'DECISIVE VICTORY'}</div>
    <div style="text-align:center;margin-bottom:1rem;">
      <div style="font-size:3rem;">${winner.emoji}</div>
      <div class="res-winner-name">${winner.name}</div>
      <div style="font-family:'Share Tech Mono',monospace;font-size:0.65rem;color:var(--nuclear);letter-spacing:3px;margin-top:0.2rem;">VICTOR</div>
    </div>
    <div class="res-analysis">${analysis}${close?' This was an exceptionally close contest — a rematch could easily go the other way.':''}</div>
    <div class="stat-compare" style="margin-top:1.5rem;">
      ${statRows.map(r=>`
        <div class="sc-stat left ${r.aWin?'sc-win':'sc-lose'}">${r.a}</div>
        <div class="sc-label">${r.label}</div>
        <div class="sc-stat right ${!r.aWin?'sc-win':'sc-lose'}">${r.b}</div>
      `).join('')}
    </div>
    <div style="display:flex;justify-content:space-between;padding-top:1rem;border-top:1px solid var(--border);margin-top:1rem;">
      <div style="font-family:'Share Tech Mono',monospace;font-size:0.65rem;color:var(--muted);">${a.name}</div>
      <div style="font-family:'Share Tech Mono',monospace;font-size:0.65rem;color:var(--muted);">${b.name}</div>
    </div>
  `;
  res.classList.add('show');
}
